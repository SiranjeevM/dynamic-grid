namespace ChartApi.Repositories;
using Microsoft.Data.SqlClient;
using ChartApi.Models;


public class ChartRepository
{
private readonly IConfiguration _configuration;

public ChartRepository(IConfiguration configuration)
{
    _configuration = configuration;
}

public List<ChartNodeDto> GetRootData()
{
    List<ChartNodeDto> result = new List<ChartNodeDto>();
    string? connectionString = _configuration.GetConnectionString("DefaultConnection");
    
    if (string.IsNullOrEmpty(connectionString))
        return result;

    using (SqlConnection con = new SqlConnection(connectionString))
    {
        con.Open();
        SqlCommand cwd = new SqlCommand(@"SELECT CategoryId, CategoryName, SalesAmount FROM Categories
            WHERE ParentCategoryID IS NULL", con);

        SqlDataReader reader = cwd.ExecuteReader();
        while (reader.Read())
        {
            result.Add(new ChartNodeDto
            {
                Id = Convert.ToInt32(reader["CategoryId"]),
                Label = reader["CategoryName"].ToString(),
                Value = Convert.ToDecimal(reader["SalesAmount"]),
            });
        }
    }
    return result;
}

public List<ChartNodeDto> GetChilder(int parentId)
{
    List<ChartNodeDto> result = new List<ChartNodeDto>();
    string? connectionString = _configuration.GetConnectionString("DefaultConnection");
    
    if (string.IsNullOrEmpty(connectionString))
        return result;

    using (SqlConnection con = new SqlConnection(connectionString))
    {
        con.Open();
        SqlCommand cwd = new SqlCommand(@"SELECT CategoryId, CategoryName, SalesAmount FROM Categories
            WHERE ParentCategoryID = @ParentId", con);
        cwd.Parameters.AddWithValue("@ParentId", parentId);

        SqlDataReader reader = cwd.ExecuteReader();
        while (reader.Read())
        {
            result.Add(new ChartNodeDto
            {
                Id = Convert.ToInt32(reader["CategoryId"]),
                Label = reader["CategoryName"].ToString(),
                Value = Convert.ToDecimal(reader["SalesAmount"]),
            });
        }
    }
    return result;
}
}

import { Injectable } from "@angular/core";
import { PieChartData, PieSlice } from "../pie-chart/pie-chart.model";

const COLORS = [
    '#FF6384','#36A2EB','#FFCE56','#4BC0C0','#9966FF','#FF9F40'
]

@Injectable({providedIn: 'root'})
export class PieChartService {

    private readonly numericKeys = ['value', 'amount', 'count', 'total', 'size'];
    private readonly labelKeys = ['label', 'name', 'code', 'category', 'title','product_name'];

    getPoint(cx:number,cy:number,radius:number,angleDeg:number){
        const rad = (angleDeg - 90) * Math.PI / 180.0;   //90 deg so that it will start from 0 degree at 12 o'clock position
        return {
            x: cx + radius * Math.cos(rad),
            y: cy + radius * Math.sin(rad)
        }
    }

    buildPath(startAngle:number,endAngle:number):string{
        const cx = 130 , cy =130;
        const outerR=100;
        const sweepAngle = endAngle - startAngle;
        if (sweepAngle >= 359.999) {
            return [
                `M ${cx} ${cy}`,
                `m 0 -${outerR}`,
                `a ${outerR} ${outerR} 0 1 1 0 ${outerR * 2}`,
                `a ${outerR} ${outerR} 0 1 1 0 -${outerR * 2}`,
                'Z'
            ].join(' ');
        }

        const startOuter = this.getPoint(cx,cy,outerR,startAngle);
        const endOuter = this.getPoint(cx,cy,outerR,endAngle);

        const large = sweepAngle > 180 ? 1 : 0; //if slice>180 degree then largearcflag will be 1 otherwise 0

        return[
            `M ${cx} ${cy}`,//move to center
            `L ${startOuter.x} ${startOuter.y}`,//line to outeredge
            `A ${outerR} ${outerR} 0 ${large} 1 ${endOuter.x} ${endOuter.y}`,//outerarc (clockwise)
            'Z'
        ].join(' ');
    }

    buildSlices(data:PieChartData[]):PieSlice[]{
        const total = data.reduce((sum, item) => sum + item.value, 0);
        if(total === 0) return [];

        let startAngle = 0;
        return data.map((item, index) => {
            const percentage = item.value / total;
            const sweep = percentage * 360;
            const endAngle = startAngle + sweep;

            const slice: PieSlice = {
                id: item.id,
                label: item.label,
                value: item.value,
                percentage,
                color: COLORS[index % COLORS.length],
                startAngle,
                endAngle,
                path: this.buildPath(startAngle, endAngle)
            };
            startAngle = endAngle;
            return slice;
        });
    }

    parseJSON(text:string):PieChartData[]{
        const raw = JSON.parse(text);
        const arraySource = this.getArraySource(raw);

        if(arraySource){
            return this.parseArrayData(arraySource);
        }

        return Object.entries(raw as Record<string, unknown>)
            .map(([label,value]) => ({
                label: String(label).trim(),
                value: parseFloat(String(value))
            }))
            .filter(d => d.label && Number.isFinite(d.value));
    }

    private getArraySource(raw: unknown): any[] | null {
        if (Array.isArray(raw)) {
            return raw;
        }

        if (!raw || typeof raw !== 'object') {
            return null;
        }

        for (const value of Object.values(raw as Record<string, unknown>)) {
            if (Array.isArray(value)) {
                return value;
            }
        }

        return null;
    }

    private parseArrayData(items: any[]): PieChartData[] {
        const hasExplicitNumericValues = items.some(item => this.getNumericValue(item) !== null);

        if (!hasExplicitNumericValues) {
            const groupBy = this.getGroupingKey(items);
            if (groupBy) {
                return this.groupAndCount(items, groupBy);
            }
        }

        const preferredLabelKey = this.getPreferredLabelKey(items);
        const preferredValueKey = this.getPreferredValueKey(items);

        const parsed = items
            .map(item => {
                const label = this.getLabel(item, preferredLabelKey);
                const numericValue = this.getNumericValue(item, preferredValueKey);

                return {
                    label,
                    value: numericValue ?? 1
                };
            })
            .filter(d => d.label && Number.isFinite(d.value) && d.value > 0);

        return this.aggregateByLabel(parsed);
    }

    private getLabel(item: any, preferredKey?: string): string {
        if (preferredKey) {
            const preferredValue = item?.[preferredKey];
            if (preferredValue !== undefined && preferredValue !== null && String(preferredValue).trim()) {
                return String(preferredValue).trim();
            }
        }

        for (const key of this.labelKeys) {
            const value = item?.[key];
            if (value !== undefined && value !== null && String(value).trim()) {
                return String(value).trim();
            }
        }

        return '?';
    }

    private getNumericValue(item: any, preferredKey?: string): number | null {
        if (preferredKey) {
            const preferredValue = item?.[preferredKey];
            const parsedPreferred = parseFloat(String(preferredValue));
            if (Number.isFinite(parsedPreferred)) {
                return parsedPreferred;
            }
        }

        for (const key of this.numericKeys) {
            const value = item?.[key];
            const parsed = parseFloat(String(value));
            if (Number.isFinite(parsed)) {
                return parsed;
            }
        }

        return null;
    }

    private getGroupingKey(items: any[]): string | null {
        if (items.some(item => item?.country)) {
            return 'country';
        }

        if (items.some(item => item?.city)) {
            return 'city';
        }

        return null;
    }

    private getPreferredLabelKey(items: any[]): string | undefined {
        const candidateKeys = ['category', 'type', 'status', 'name', 'label', 'code', 'title'];
        for (const key of candidateKeys) {
            const coverage = items.filter(item => item?.[key] !== undefined && item?.[key] !== null && String(item[key]).trim()).length;
            if (coverage >= Math.ceil(items.length * 0.7)) {
                return key;
            }
        }

        return undefined;
    }

    private getPreferredValueKey(items: any[]): string | undefined {
        for (const key of this.numericKeys) {
            const coverage = items.filter(item => Number.isFinite(parseFloat(String(item?.[key])))).length;
            if (coverage >= Math.ceil(items.length * 0.7)) {
                return key;
            }
        }

        return undefined;
    }

    private aggregateByLabel(rows: PieChartData[]): PieChartData[] {
        const sums = new Map<string, number>();

        for (const row of rows) {
            sums.set(row.label, (sums.get(row.label) ?? 0) + row.value);
        }

        return Array.from(sums.entries())
            .map(([label, value]) => ({ label, value }))
            .sort((a, b) => b.value - a.value);
    }

    private groupAndCount(items: any[], key: string): PieChartData[] {
        const counts = new Map<string, number>();

        for (const item of items) {
            const label = String(item?.[key] ?? '').trim();
            if (!label) {
                continue;
            }
            counts.set(label, (counts.get(label) ?? 0) + 1);
        }

        return Array.from(counts.entries())
            .map(([label, value]) => ({ label, value }))
            .sort((a, b) => b.value - a.value);
    }
}

import { CommonModule } from '@angular/common';
import { Component, OnInit } from '@angular/core';
import { ChartApiService } from './services/chart-api.service';
import { PieChartComponent } from './pie-chart/pie-chart.component';
import { BarChartComponent } from './bar-chart/bar-chart.component';
import { ChartNode } from './models/chart-node.model';

@Component({
  selector: 'app-root',
  standalone: true,
  templateUrl: './app.component.html',
  styleUrls: ['./app.component.scss'],
  imports: [CommonModule, PieChartComponent, BarChartComponent]
})
export class AppComponent implements OnInit {
  currentData: ChartNode[] = [];
  breadcrumbs: string[] = [];
  history: ChartNode[][] = [];
  rootData: ChartNode[] = [];

  currentChartType: 'pie' | 'bar' = 'pie';
  level = 0;

  constructor(private chartApi: ChartApiService) {}

  ngOnInit(): void {
    this.chartApi.getRootChart().subscribe(result => {
      this.currentData = result;
      this.rootData = result;
      this.level = 0;
      this.currentChartType = 'pie';
    });
  }

  onNodeSelected(node: ChartNode): void {
    if (!node.id) {
      return;
    }

    this.chartApi.getDrilldown(node.id).subscribe(result => {
      if (!result.length) {
        return;
      }

      this.history.push(this.currentData);
      this.breadcrumbs.push(node.label);
      this.currentData = result;
      this.level++;
      this.currentChartType = this.level % 2 === 0 ? 'pie' : 'bar';
    });
  }

  goBack(): void {
    if (this.history.length === 0) {
      return;
    }
    this.currentData = this.history.pop()!;
    this.breadcrumbs.pop();
    this.level--;
    this.currentChartType = this.level % 2 === 0 ? 'pie' : 'bar';
  }

  goHome(): void {
    this.currentData = this.rootData;
    this.history = [];
    this.breadcrumbs = [];
    this.level = 0;
    this.currentChartType = 'pie';
  }

  navigateUp(): void {
    if (this.history.length > 1) {
      this.goBack();
      return;
    }
    this.goHome();
  }
}

  ngOnChanges(): void {
    this.hoveredIndex = -1;
    this.maxValue = this.data.length ? Math.max(...this.data.map(d => d.value)) : 0;
  }

  onBarClicked(bar: BarChartData): void {
    this.barClicked.emit(bar);
  }

  onHover(index: number): void {
    this.hoveredIndex = index;
  }

  onLeave(): void {
    this.hoveredIndex = -1;
  }

  getColor(index: number): string {
    return COLORS[index % COLORS.length];
  }

  getWidth(value: number): number {
    if (this.maxValue <= 0) {
      return 0;
    }
    return (value / this.maxValue) * 100;
  }

  getTotal(): number {
    return this.data.reduce((sum, item) => sum + item.value, 0);
  }

  getPercentage(value: number): number {
    const total = this.getTotal();
    return total > 0 ? value / total : 0;
  }

  toPct(value: number): string {
    return (value * 100).toFixed(1) + '%';
  }
}
